#We consider the monthly sales of products (see below for file SWK-14.csv).
#We download the data
sales = read.csv("SWK-14.csv")

#we check for NAs, we find none
sum(is.na(sales[,2]) == TRUE)
## [1] 0
#we transform the data into time series and plot it 
sales_ts = ts(sales$Sales, start = 2015, frequency = 12)

par(mfrow=c(1,1))

ts.plot(sales_ts, col = "navy", main = "SALES",
        gpars = list(xlab = "year", ylab = "sales"))

adf.test(sales_ts)
## 
##  Augmented Dickey-Fuller Test
## 
## data:  sales_ts
## Dickey-Fuller = -1.3547, Lag order = 5, p-value = 0.8448
## alternative hypothesis: stationary
#we can't reject the null hypothesis of data nonstationarity
#we notice a downward trend starting circa 2021, we remove it 

diff_ts = diff(sales_ts)
plot(diff_ts,col = "navy", main = "SALES after diff",
     gpars = list(xlab = "year", ylab = "sales"))

#we notice an oscillation around mean 0 which might suggest stationarity 

#adf.test tells us that there indeed is stationarity
adf.test(diff_ts)
## 
##  Augmented Dickey-Fuller Test
## 
## data:  diff_ts
## Dickey-Fuller = -5.1819, Lag order = 5, p-value = 0.01
## alternative hypothesis: stationary
par(mfrow=c(1,2))
acf(diff_ts,  main=" ",lag.max=36, ci=.95)
pacf(diff_ts, main=" ",lag.max=36, ci=.95)

#We notice that:
# - ACF tails off → consistent with AR process
# - PACF has significant value at lag 1, possibly at lag 2 → suggests AR(1) or AR(2)
# - The one PACF value near lag 3 slightly exceeds the confidence interval, 
#   we presume it to be a single outlier without great relevance to the data as a whole
# - Following the table: ACF tails off + PACF cuts off after lag p → AR(p) model
# - We will consider AR(1) and AR(2) for the differenced series

#We plot periodogram for our data
n = length(diff_ts)
Per= Mod(fft(diff_ts-mean(diff_ts)))^2/n  
Freq = (1:n -1)/n

par(mfrow=c(1,1))
plot(Freq[1:n], Per[1:n], type='h', lwd=2, ylab="Periodogram",
     xlab="Frequency")  

nn=n/2

u = which.max(Per[1:nn])
uu = which.max(Per[1:nn][-u])
1/Freq[u]
